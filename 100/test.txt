int time_order(double time[100], char name[100][32], char school[100][32], int grade[100], int day[100]){
	int i;
	double mem[100];
    for(i = 0; i < 100; ++i){
        mem[i] = time[i];
    }
	int j,k;
    double tmp;
    for(j=0; j <= 100; ++j){
        for(k = j + 1; k <= 100; ++k){
            if(time[j] > time[k]){
                tmp = time[j];
                time[j] = time[k];
                time[k] = tmp;
            }
        }
    }
	printf("<タイム順>\n");
    for(j=0; j<100; ++j){
        if(time[j] > 5  && time[j] < 50 && time[j] != time[j-1]){
            printf("タイム:%.2lf秒\n", time[j]);
                for(i=0; i <= 100; ++i){
                    if(time[j] == mem[i]){
                    printf("名前:%s 学校:%s 学年:%d年 記録日:%d日\n",name[i],school[i],grade[i],day[i]);
                }
            }
            printf("\n");
        }
        
        
        
    }
	return 0;
}
